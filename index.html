// ============================================================================= // BRILIA — Dashboard KPI 2026 — v4 // Inclui: Pareto de Defeitos + Tempo Médio de Separação // ============================================================================= const SHEET_DASHBOARD = 'DASHBOARD'; const SHEET_SCRAP_RAW = 'SCRAP_RAW'; const SHEET_SAP_RAW = 'SAP_RAW'; const SHEET_BOM_RAW = 'BOM_RAW'; const ID_REQUISICOES = '1B8rZs4K-WRBOIFa1ocrrBrkjRKhg3g0H0HR0iYOKqio'; const ABA_REQUISICOES = 'Requisicoes'; const MESES = ['Jan','Fev','Mar','Abr','Mai','Jun','Jul','Ago','Set','Out','Nov','Dez']; // ============================================================================= // doGet — retorna JSON completo para o dashboard do CEO // ============================================================================= function doGet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const dash = ss.getSheetByName(SHEET_DASHBOARD); const data = dash.getDataRange().getValues(); // Encontra linha inicial (Jan) let headerRow = -1; for (let i = 0; i < data.length; i++) { const cell = String(data[i][0]).toLowerCase(); if (cell.includes('jan') || cell.includes('fev')) { headerRow = i; break; } } if (headerRow === -1) { return ContentService .createTextOutput(JSON.stringify({ erro: 'Linha Jan não encontrada' })) .setMimeType(ContentService.MimeType.JSON); } // Calcula Pareto por mês const paretoPorMes = calcularParetoPorMes(); // Calcula Tempo de Separação por mês const tempoPorMes = calcularTempoPorMes(); const meses = ['jan','fev','mar','abr','mai','jun','jul','ago','set','out','nov','dez']; const rows = []; for (let i = headerRow; i < data.length; i++) { const row = data[i]; const mes = String(row[0]).toLowerCase(); const mesIdx = meses.findIndex(m => mes.includes(m)); if (mesIdx === -1) continue; rows.push({ mes: String(row[0]), scrapQty: Number(row[1]) || 0, scrapCost: Number(row[2]) || 0, bomErros: Number(row[3]) || 0, minPlan: Number(row[4]) || 0, minReal: Number(row[5]) || 0, efic: Number(row[6]) || 0, pareto: paretoPorMes[mesIdx] || null, tempoSep: tempoPorMes[mesIdx] || null, }); } return ContentService .createTextOutput(JSON.stringify(rows)) .setMimeType(ContentService.MimeType.JSON); } // ============================================================================= // calcularParetoPorMes — conta defeitos por mês da SCRAP_RAW // ============================================================================= function calcularParetoPorMes() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const wsRaw = ss.getSheetByName(SHEET_SCRAP_RAW); if (!wsRaw) return {}; const data = wsRaw.getDataRange().getValues(); const result = {}; // { mesIdx: { defeito: count } } for (let i = 1; i < data.length; i++) { const dateVal = data[i][0]; const defeito = String(data[i][3]).trim(); // Col D = Defeito const qty = Number(data[i][4]) || 1; // Col E = QTD if (!dateVal || !defeito || defeito === '') continue; let date; try { date = new Date(dateVal); if (isNaN(date.getTime())) continue; } catch(e) { continue; } if (date.getFullYear() !== 2026) continue; const m = date.getMonth(); if (!result[m]) result[m] = {}; result[m][defeito] = (result[m][defeito] || 0) + qty; } return result; } // ============================================================================= // calcularTempoPorMes — calcula tempo médio de separação por mês // Col L = Timestamp_Separacao, Col M = Timestamp (criação) // ============================================================================= function calcularTempoPorMes() { try { const ssReq = SpreadsheetApp.openById(ID_REQUISICOES); const wsReq = ssReq.getSheetByName(ABA_REQUISICOES); if (!wsReq) return {}; const data = wsReq.getDataRange().getValues(); const header = data[0].map(h => String(h).toLowerCase().trim()); // Encontra índices das colunas const colData = header.indexOf('data'); const colTsSep = header.findIndex(h => h.includes('timestamp_sep') || h.includes('separacao')); const colTs = header.findIndex(h => h === 'timestamp' || h.includes('timestamp') && !h.includes('sep')); if (colTsSep === -1 || colTs === -1) return {}; const result = {}; // { mesIdx: { tempos: [], total: n } } for (let i = 1; i < data.length; i++) { const dateVal = data[i][colData]; const tsSep = data[i][colTsSep]; const tsCreate = data[i][colTs]; if (!dateVal || !tsSep || !tsCreate) continue; let date, dateSep, dateCreate; try { date = new Date(dateVal); dateSep = new Date(tsSep); dateCreate = new Date(tsCreate); if (isNaN(date.getTime()) || isNaN(dateSep.getTime()) || isNaN(dateCreate.getTime())) continue; } catch(e) { continue; } if (date.getFullYear() !== 2026) continue; const diffHoras = (dateSep - dateCreate) / (1000 * 60 * 60); if (diffHoras < 0 || diffHoras > 720) continue; // ignora valores absurdos const m = date.getMonth(); if (!result[m]) result[m] = { tempos: [], total: 0 }; result[m].tempos.push(diffHoras); result[m].total++; } // Calcula médias const summary = {}; Object.entries(result).forEach(([m, d]) => { if (d.tempos.length === 0) return; const media = d.tempos.reduce((a,v)=>a+v,0) / d.tempos.length; summary[m] = { media: parseFloat(media.toFixed(2)), min: parseFloat(Math.min(...d.tempos).toFixed(2)), max: parseFloat(Math.max(...d.tempos).toFixed(2)), total: d.total, }; }); return summary; } catch(e) { return {}; } } // ============================================================================= // atualizarScrap // ============================================================================= function atualizarScrap() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let wsRaw = ss.getSheetByName(SHEET_SCRAP_RAW); if (!wsRaw) { wsRaw = ss.insertSheet(SHEET_SCRAP_RAW); wsRaw.getRange('A1:G1').setValues([['Data','SKU','Descrição','Defeito','QTD','CustoUn','CustoTotal']]); SpreadsheetApp.getUi().alert('Aba SCRAP_RAW criada!\nCole os dados a partir da linha 2 e execute novamente.'); return; } const data = wsRaw.getDataRange().getValues(); if (data.length <= 1) { SpreadsheetApp.getUi().alert('SCRAP_RAW está vazia!'); return; } const scrapQTD = new Array(12).fill(0); const scrapCUST = new Array(12).fill(0); for (let i = 1; i < data.length; i++) { const dateVal = data[i][0]; if (!dateVal) continue; let date; try { date = new Date(dateVal); if (isNaN(date.getTime())) continue; } catch(e) { continue; } if (date.getFullYear() !== 2026) continue; const m = date.getMonth(); scrapQTD[m] += Number(data[i][4]) || 0; const custoStr = String(data[i][6]).replace('R$','').replace(/\./g,'').replace(',','.').trim(); scrapCUST[m] += Number(custoStr) || 0; } atualizarDashboard('scrap', scrapQTD, scrapCUST); SpreadsheetApp.getUi().alert('✅ Scrap atualizado!\n\n' + MESES.map((m,i) => `${m}: ${scrapQTD[i].toFixed(0)} pçs | R$ ${scrapCUST[i].toFixed(2)}`).join('\n')); } // ============================================================================= // atualizarMinutos // ============================================================================= function atualizarMinutos() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let wsRaw = ss.getSheetByName(SHEET_SAP_RAW); if (!wsRaw) { wsRaw = ss.insertSheet(SHEET_SAP_RAW); wsRaw.getRange('A1:K1').setValues([['ClasseCusto','Denominação','Valor/MR','QtdMIN','UnMed','F','G','H','I','J','Data']]); SpreadsheetApp.getUi().alert('Aba SAP_RAW criada!\nCole a exportação do SAP a partir da linha 2 e execute novamente.'); return; } const data = wsRaw.getDataRange().getValues(); if (data.length <= 1) { SpreadsheetApp.getUi().alert('SAP_RAW está vazia!'); return; } let totalMin = 0; let mesIdx = -1; for (let i = data.length - 1; i >= 1; i--) { if (totalMin === 0 && data[i][3] && !String(data[i][0]).includes('Ordem')) { const valStr = String(data[i][3]).replace(/\./g,'').replace(',','.').trim(); totalMin = Number(valStr) || 0; } if (mesIdx === -1 && data[i][10]) { const date = new Date(data[i][10]); if (!isNaN(date.getTime()) && date.getFullYear() === 2026) mesIdx = date.getMonth(); } if (totalMin > 0 && mesIdx !== -1) break; } if (mesIdx === -1) { SpreadsheetApp.getUi().alert('Datas não encontradas na coluna K.'); return; } const minReal = new Array(12).fill(0); minReal[mesIdx] = totalMin; atualizarDashboard('minutos', minReal); SpreadsheetApp.getUi().alert(`✅ Minutos atualizados!\n\n${MESES[mesIdx]}: ${totalMin.toLocaleString()} min`); } // ============================================================================= // atualizarBOMdoRequisicoes // ============================================================================= function atualizarBOMdoRequisicoes() { const ssReq = SpreadsheetApp.openById(ID_REQUISICOES); const wsReq = ssReq.getSheetByName(ABA_REQUISICOES); if (!wsReq) { SpreadsheetApp.getUi().alert('Aba Requisicoes não encontrada.'); return; } const data = wsReq.getDataRange().getValues(); const header = data[0].map(h => String(h).toLowerCase().trim()); const colData = header.indexOf('data'); const colMotivo = header.indexOf('motivo'); if (colData === -1 || colMotivo === -1) { SpreadsheetApp.getUi().alert('Colunas não encontradas.'); return; } const bomErros = new Array(12).fill(0); let total = 0; for (let i = 1; i < data.length; i++) { const motivo = String(data[i][colMotivo]).toLowerCase().trim(); const dateVal = data[i][colData]; if (!motivo.includes('bom') || !dateVal) continue; let date; try { date = new Date(dateVal); if (isNaN(date.getTime())) continue; } catch(e) { continue; } if (date.getFullYear() !== 2026) continue; bomErros[date.getMonth()]++; total++; } atualizarDashboard('bom', bomErros); const resumo = total === 0 ? 'Nenhum BOM Errada encontrado.' : MESES.map((m,i) => bomErros[i]>0 ? `${m}: ${bomErros[i]} erros` : null).filter(Boolean).join('\n'); SpreadsheetApp.getUi().alert(`✅ BOM atualizado!\n\nTotal: ${total} ocorrências\n\n${resumo}`); } // ============================================================================= // atualizarMinPlan — proporcional aos dias úteis // ============================================================================= function atualizarMinPlan() { const ui = SpreadsheetApp.getUi(); const ss = SpreadsheetApp.getActiveSpreadsheet(); const wsDash = ss.getSheetByName(SHEET_DASHBOARD); const resp = ui.prompt('Minutos Planejados', 'Digite: mês, minutos planejados, dias úteis do mês\nEx: 5,70896,21', ui.ButtonSet.OK_CANCEL); if (resp.getSelectedButton() !== ui.Button.OK) return; const input = resp.getResponseText().split(','); if (input.length !== 3) { ui.alert('Formato inválido. Ex: 5,70896,21'); return; } const mesNum = parseInt(input[0].trim()) - 1; const minTotal = parseFloat(input[1].trim()); const diasUteis = parseInt(input[2].trim()); if (mesNum < 0 || mesNum > 11 || isNaN(minTotal) || isNaN(diasUteis)) { ui.alert('Valores inválidos.'); return; } const hoje = new Date(); const ultimoDia = mesNum === hoje.getMonth() ? hoje.getDate() : new Date(hoje.getFullYear(), mesNum+1, 0).getDate(); let diasDecorridos = 0; for (let d = 1; d <= ultimoDia; d++) { const dw = new Date(hoje.getFullYear(), mesNum, d).getDay(); if (dw !== 0 && dw !== 6) diasDecorridos++; } const minProporcional = Math.round(minTotal * diasDecorridos / diasUteis); const data = wsDash.getDataRange().getValues(); const mesRows = encontrarLinhasMeses(data); const rowIdx = mesRows[mesNum]; if (rowIdx !== -1) { wsDash.getRange(rowIdx+1, 5).setValue(minProporcional); const nd = wsDash.getDataRange().getValues(); recalcularEficiencia(wsDash, nd, encontrarLinhasMeses(nd)); } ui.alert(`✅ Min Planejados atualizados!\n\n${MESES[mesNum]}: ${minTotal.toLocaleString()} min\nDias úteis: ${diasUteis} | Decorridos: ${diasDecorridos}\nMeta proporcional: ${minProporcional.toLocaleString()} min`); } // ============================================================================= // atualizarTudo // ============================================================================= function atualizarTudo() { atualizarScrap(); atualizarMinutos(); atualizarBOMdoRequisicoes(); SpreadsheetApp.getUi().alert('✅ Dashboard atualizado completamente!'); } // ============================================================================= // FUNÇÕES AUXILIARES // ============================================================================= function atualizarDashboard(tipo, arr1, arr2) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const wsDash = ss.getSheetByName(SHEET_DASHBOARD); const data = wsDash.getDataRange().getValues(); const mesRows = encontrarLinhasMeses(data); for (let i = 0; i < 12; i++) { const ri = mesRows[i]; if (ri === -1) continue; if (tipo === 'scrap') { wsDash.getRange(ri+1, 2).setValue(arr1[i]); wsDash.getRange(ri+1, 3).setValue(arr2[i]); } else if (tipo === 'minutos') { wsDash.getRange(ri+1, 6).setValue(arr1[i]); } else if (tipo === 'bom') { wsDash.getRange(ri+1, 4).setValue(arr1[i]); } } const nd = wsDash.getDataRange().getValues(); recalcularEficiencia(wsDash, nd, encontrarLinhasMeses(nd)); } function encontrarLinhasMeses(data) { const ml = ['jan','fev','mar','abr','mai','jun','jul','ago','set','out','nov','dez']; const r = new Array(12).fill(-1); for (let i = 0; i < data.length; i++) { const cell = String(data[i][0]).toLowerCase().trim(); for (let m = 0; m < 12; m++) { if (cell.includes(ml[m])) { r[m] = i; break; } } } return r; } function recalcularEficiencia(wsDash, data, mesRows) { for (let i = 0; i < 12; i++) { const ri = mesRows[i]; if (ri === -1) continue; const mp = Number(data[ri][4]) || 0; const mr = Number(data[ri][5]) || 0; wsDash.getRange(ri+1, 7).setValue(mp > 0 ? mr/mp : 0); } } // ============================================================================= // onOpen — menu // ============================================================================= function onOpen() { SpreadsheetApp.getUi() .createMenu('🔄 BRILIA Dashboard') .addItem('📦 Atualizar Scrap', 'atualizarScrap') .addItem('⏱ Atualizar Minutos', 'atualizarMinutos') .addItem('📋 BOM do Sistema de Desvios', 'atualizarBOMdoRequisicoes') .addItem('📅 Definir Min Planejados', 'atualizarMinPlan') .addSeparator() .addItem('🔄 Atualizar Tudo', 'atualizarTudo') .addToUi(); }
